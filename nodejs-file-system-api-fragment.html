Node.js 文件系统(fs) API代码片段

官网文档：https://nodejs.org/api/fs.html

fs.readFile(file[, options], callback)

var fs = require('fs');
fs,readFile('path/to/file', 'utf8', function(err, data) {
  if (err) throw err;
  console.log(data.toString('utf8'));
  // data 为 Buffer
  // eg: <Buffer 65 6d 61 69 6c 20 3d 0a>
})
