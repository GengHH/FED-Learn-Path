    .insertAdjacentHTML('afterend', '<div id="insertBefore">insertAfter</div>');

</div>

// movingBefore
document.getElementById('moveParent')
    .insertBefore(document.getElementById('moveElement'),
        document.getElementById('moveFirst'));
// movingAfter
document.getElementById('moveParent')
    .appendChild(document.getElementById('moveElement'));

添加 & 移除类 (Adding & Removing Classes)

<style> .class-foo {font-weight: 600;} </style>
<div id="classElement">样式在手，天下我有</div> 
// Add Class
document.getElementById('classElement').className += 'class-foo';
// Remove Class
var classElem = document.getElementById('classElement');
classElem.className =
    classElem.className.replace(/class-foo/, '');

// Change Text
var textElem = document.getElementById('textElement');
// 修改text会自动HTML转义内容
if (textElem.textContent) {
    // Need IE9+
    textElem.textContent = '<div>textContent</div>';
} else {
    // Firefox not support
    textElem.innerText = '<div>innerText</div>';
}

获取/添加/移除元素属性 (Getting/Adding/Removing Element Attributes) [详细]

<button id="attrBtn" role="button">我是个按钮</button> 

// getAttribute
document.getElementById('attrBtn').getAttribute('role');
// setAttribute
document.getElementById('attrBtn').setAttribute('disabled', '');
// removeAttribute
document.getElementById('attrBtn').removeAttribute('disabled');
