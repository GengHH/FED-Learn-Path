元素前后插入 (Inserting Elements Before & After)

<div id="insertElem1">insertElem1</div>
<div id="insertElem2">insertElem2</div>
<div id="insertElem3">insertElem3</div> 
// insertChildBefore
document.getElementById('insertElem2')
    .insertAdjacentHTML('beforebegin', '<div id="insertBefore">insertBefore</div>');
// insertChildAfter
document.getElementById('insertElem2')
    .insertAdjacentHTML('afterend', '<div id="insertBefore">insertAfter</div>');

移动元素 (Moving Elements)

<div id="moveParent">

<div id="moveFirst">moveChild</div>
<div>moveChild</div>
<div>moveChild</div>

</div>

<div id="moveElement">被移动元素</div> 
// movingBefore
document.getElementById('moveParent')
    .insertBefore(document.getElementById('moveElement'),
        document.getElementById('moveFirst'));
// movingAfter
document.getElementById('moveParent')
    .appendChild(document.getElementById('moveElement'));

    .removeChild(removeElem);

添加 & 移除类 (Adding & Removing Classes)

<style> .class-foo {font-weight: 600;} </style>
<div id="classElement">样式在手，天下我有</div> 
// Add Class
document.getElementById('classElement').className += 'class-foo';
// Remove Class
var classElem = document.getElementById('classElement');
classElem.className =
    classElem.className.replace(/class-foo/, '');
