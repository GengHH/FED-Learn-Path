创建元素 (Creating Elements)

    var div = document.createElement('div');

元素前后插入 (Inserting Elements Before & After)

<div id="insertElem1">insertElem1</div>
<div id="insertElem2">insertElem2</div>
<div id="insertElem3">insertElem3</div> 
    // insertChildBefore
    document.getElementById('insertElem2')
        .insertAdjacentHTML('beforebegin', '<div id="insertBefore">insertBefore</div>');
    // insertChildAfter
    document.getElementById('insertElem2')
        .insertAdjacentHTML('afterend', '<div id="insertBefore">insertAfter</div>');

插入子元素 (Inserting Elements As Children)

<div id="oldChild">oldChild</div> 
    // insertBefore
    document.getElementById('parent')
        .insertAdjacentHTML('afterbegin', '<div id="insertBefore">insertBefore</div>');
    // insertAfter
    document.getElementById('insertElem2')
        .insertAdjacentHTML('beforeend', '<div id="insertBefore">insertAfter</div>');
