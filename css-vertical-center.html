float & clear 垂直居中

浮动元素

定高

    /** HTML **/
    <div class="float-wrapper" style="border: 1px solid #38D">
        <div class="float-node"></div>
        <div class="float-center" style="background: #f5f4f3">
            <p>浮动元素</p>
            <p>定高</p>
        </div>
    </div>
    /** CSS **/
    <style>
    .float-wrapper {
        height: 300px;
    }
    .float-node {
        height: 500px;
        float: left;
        height: 50%;
        margin-bottom: -60px;
    }
    .float-center {
        position: relative;
        width: 140px;
        height: 120px;
        margin: 0 auto;
        clear: both;
    }
    </style>
